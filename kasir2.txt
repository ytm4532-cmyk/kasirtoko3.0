  <script>
    let daftarProdukMaster = [];
    let keranjangBelanja = [];
    let currentTrxId = "";
    let currentTotalBelanja = 0;
    let html5QrcodeScanner = null;
    let isScanning = false;
    let lastScannedCode = "";
    let lastScannedTime = 0;

    document.addEventListener("DOMContentLoaded", function() { muatDataAwal(); });

    function muatDataAwal() {
      showLoadingModal("Sinkronisasi", "Menghubungkan ke database Google Sheets...");
      fetch(`${API_URL}?action=getTrxId`)
        .then(res => res.json())
        .then(res => {
          currentTrxId = res.trxId;
          document.getElementById("display-trx-id").innerText = res.trxId;
          return fetch(`${API_URL}?action=getProduk`);
        })
        .then(res => res.json())
        .then(data => {
          daftarProdukMaster = data;
          isiDropdownProduk(data);
          tutupModal();
        })
        .catch(err => { 
          console.error(err);
          showErrorModal("Error Koneksi", "Gagal mengambil data dari Google Sheets. Periksa Deploy & Nama Tab!"); 
        });
    }

    function isiDropdownProduk(list) {
      const datalist = document.getElementById("daftar-produk-datalist");
      datalist.innerHTML = ''; 
      
      list.forEach(p => {
        if(p.stok > 0) {
          let opt = document.createElement("option");
          opt.value = p.id; 
          opt.innerText = `${p.nama} (Stok: ${p.stok})`;
          datalist.appendChild(opt);
        }
      });
    }

    function updateDetailProdukPilihan() {
      const inputVal = document.getElementById("select-produk").value.trim();
      const p = daftarProdukMaster.find(prod => prod.id === inputVal);
      
      document.getElementById("info-harga").innerText = p ? formatRupiah(p.harga) : "Rp 0";
      document.getElementById("info-stok").innerText = p ? p.stok : "0";
    }

    function toggleScanner() {
      const btn = document.getElementById("btn-toggle-scan");
      if (isScanning) {
        html5QrcodeScanner.stop().then(() => {
          document.getElementById("reader").innerHTML = "Kamera dinonaktifkan";
          btn.innerText = "Aktifkan Kamera";
          isScanning = false;
        });
      } else {
        document.getElementById("reader").innerHTML = "";
        btn.innerText = "Matikan Kamera";
        isScanning = true;
        html5QrcodeScanner = new Html5Qrcode("reader");
        html5QrcodeScanner.start(
          { facingMode: "environment" }, 
          { fps: 15, qrbox: (w, h) => ({ width: w*0.8, height: h*0.5 }) },
          (text) => {
            const now = Date.now();
            const barcodeScanned = text.trim();
            
            // Mencegah double scan berturut-turut dalam waktu 2 detik untuk kode yang sama
            if (barcodeScanned === lastScannedCode && (now - lastScannedTime < 2000)) {
              return; 
            }
            
            lastScannedCode = barcodeScanned;
            lastScannedTime = now;

            if (navigator.vibrate) navigator.vibrate(100);
            
            const prod = daftarProdukMaster.find(p => p.id === barcodeScanned);
            if (prod) {
              document.getElementById("select-produk").value = prod.id;
              updateDetailProdukPilihan();
              // Khusus SCAN BARCODE: Hanya memasukkan tepat 1 Qty ke keranjang
              tambahDariBarcode(prod);
            } else { 
              alert("Barcode '" + barcodeScanned + "' tidak ditemukan di tab Produk!"); 
            }
          }
        ).catch((err) => { 
          alert("Gagal mengakses kamera. Jika pakai HP, pastikan membuka lewat link HTTPS!"); 
          toggleScanner(); 
        });
      }
    }

    // Fungsi khusus menangani input barcode agar tepat masuk 1 Qty
    function tambahDariBarcode(p) {
      let exist = keranjangBelanja.find(item => item.id === p.id);
      if(exist) {
        if(exist.qty + 1 > p.stok) { alert("Stok di Google Sheets tidak mencukupi!"); return; }
        exist.qty += 1;
        exist.subtotal = exist.qty * exist.harga;
      } else {
        keranjangBelanja.push({ id: p.id, nama: p.nama, harga: p.harga, qty: 1, subtotal: p.harga });
      }
      clearInputProduk();
      renderKeranjang();
    }

    function tambahKeKeranjang() {
      const id = document.getElementById("select-produk").value.trim();
      if(!id) return;
      const p = daftarProdukMaster.find(prod => prod.id === id);
      if(!p) { alert("Produk tidak valid!"); return; }
      
      const qty = parseInt(document.getElementById("input-qty").value) || 1;
      if(qty > p.stok) { alert("Stok tidak mencukupi!"); return; }
      
      let exist = keranjangBelanja.find(item => item.id === id);
      if(exist) {
        if(exist.qty + qty > p.stok) { alert("Total belanjaan melebihi stok tersedia!"); return; }
        exist.qty += qty; 
        exist.subtotal = exist.qty * exist.harga;
      } else {
        keranjangBelanja.push({ id: p.id, nama: p.nama, harga: p.harga, qty: qty, subtotal: qty * p.harga });
      }
      clearInputProduk();
      renderKeranjang();
    }

    function clearInputProduk() {
      document.getElementById("select-produk").value = "";
      document.getElementById("input-qty").value = "1";
      updateDetailProdukPilihan();
    }

    // FUNGSI UTAMA BARU: Tombol Plus (+) dan Minus (-) di dalam tabel keranjang
    function sesuaikanQtyKlik(id, arah) {
      let item = keranjangBelanja.find(i => i.id === id);
      const p = daftarProdukMaster.find(prod => prod.id === id);
      if (!item || !p) return;

      let qtyBaru = item.qty + arah;

      if (qtyBaru < 1) {
        alert("Jumlah minimal adalah 1! Gunakan ikon sampah di kanan jika ingin menghapus produk.");
        return;
      }

      if (qtyBaru > p.stok) {
        alert(`Stok tidak mencukupi! Batas stok di Google Sheets adalah ${p.stok}`);
        return;
      }

      item.qty = qtyBaru;
      item.subtotal = qtyBaru * item.harga;
      
      renderKeranjang();
    }

    function ubahQtyKetikManual(id, elemenInput) {
      let qtyBaru = parseInt(elemenInput.value);
      const p = daftarProdukMaster.find(prod => prod.id === id);
      
      if (isNaN(qtyBaru) || qtyBaru < 1) {
        qtyBaru = 1;
      }
      
      if (qtyBaru > p.stok) {
        alert(`Stok maksimal hanya ${p.stok}!`);
        qtyBaru = p.stok;
      }
      
      let item = keranjangBelanja.find(i => i.id === id);
      if (item) {
        item.qty = qtyBaru;
        item.subtotal = qtyBaru * item.harga;
      }
      
      currentTotalBelanja = 0;
      keranjangBelanja.forEach(i => currentTotalBelanja += i.subtotal);
      document.getElementById(`subtotal-${id}`).innerText = formatRupiah(item.subtotal);
      document.getElementById("display-total").innerText = formatRupiah(currentTotalBelanja);
      hitungKembalian();
    }

    function renderKeranjang() {
      const tbody = document.getElementById("tabel-keranjang");
      tbody.innerHTML = keranjangBelanja.length === 0 ? '<tr><td colspan="5" class="p-8 text-center text-slate-400 italic">Keranjang masih kosong</td></tr>' : '';
      currentTotalBelanja = 0;
      
      keranjangBelanja.forEach(item => {
        currentTotalBelanja += item.subtotal;
        let tr = document.createElement("tr");
        tr.innerHTML = `
          <td class="p-3 font-medium text-slate-700">${item.nama}</td>
          <td class="p-3 text-center">
            <div class="flex items-center justify-center gap-1">
              <button onclick="sesuaikanQtyKlik('${item.id}', -1)" class="w-7 h-7 bg-slate-200 hover:bg-slate-300 text-slate-700 rounded font-bold transition flex items-center justify-center cursor-pointer text-xs select-none">-</button>
              <input type="number" value="${item.qty}" 
                class="w-12 h-7 border border-slate-300 rounded text-center bg-white font-semibold text-sm focus:outline-none"
                oninput="ubahQtyKetikManual('${item.id}', this)"
                onchange="renderKeranjang()">
              <button onclick="sesuaikanQtyKlik('${item.id}', 1)" class="w-7 h-7 bg-indigo-100 hover:bg-indigo-200 text-indigo-700 rounded font-bold transition flex items-center justify-center cursor-pointer text-xs select-none">+</button>
            </div>
          </td>
          <td class="p-3 text-right text-slate-600">${formatRupiah(item.harga)}</td>
          <td class="p-3 text-right font-bold text-indigo-600" id="subtotal-${item.id}">${formatRupiah(item.subtotal)}</td>
          <td class="p-3 text-center">
            <button onclick="hapus('${item.id}')" class="text-rose-500 cursor-pointer hover:text-rose-700 p-1"><i class="fa-solid fa-trash"></i></button>
          </td>
        `;
        tbody.appendChild(tr);
      });
      document.getElementById("display-total").innerText = formatRupiah(currentTotalBelanja);
      hitungKembalian();
    }

    function hapus(id) { 
      keranjangBelanja = keranjangBelanja.filter(i => i.id !== id); 
      renderKeranjang(); 
    }

    function hitungKembalian() {
      let bayar = parseFloat(document.getElementById("input-bayar").value) || 0;
      let lbl = document.getElementById("display-kembalian");
      let metode = document.getElementById("select-metode").value;
      
      if(metode !== "Cash") { 
        document.getElementById("input-bayar").value = currentTotalBelanja; 
        lbl.innerText = "Rp 0"; 
        lbl.className = "text-xl font-mono font-bold text-emerald-600";
        return; 
      }
      
      let sisa = bayar - currentTotalBelanja;
      if (sisa < 0) {
        lbl.innerText = "Kurang " + formatRupiah(Math.abs(sisa));
        lbl.className = "text-xl font-mono font-bold text-rose-500";
      } else {
        lbl.innerText = formatRupiah(sisa);
        lbl.className = "text-xl font-mono font-bold text-emerald-600";
      }
    }

    document.getElementById("select-metode").addEventListener("change", hitungKembalian);

    function prosesSelesaiTransaksi() {
      if(keranjangBelanja.length === 0) return;
      let bayar = parseFloat(document.getElementById("input-bayar").value) || 0;
      let metode = document.getElementById("select-metode").value;
      if(metode === "Cash" && bayar < currentTotalBelanja) { alert("Uang pembayaran kurang!"); return; }

      let payload = { 
        idTransaksi: currentTrxId, 
        totalBelanja: currentTotalBelanja, 
        metodePembayaran: metode, 
        uangBayar: bayar, 
        uangKembalian: metode === "Cash" ? bayar - currentTotalBelanja : 0, 
        cart: keranjangBelanja 
      };
      
      showLoadingModal("Menyimpan", "Mengirim data transaksi ke Google Sheets...");

      fetch(API_URL, { method: "POST", body: JSON.stringify(payload) })
        .then(res => res.json())
        .then(res => {
          if(res.success) {
            showSuccessModal("Berhasil!", res.message);
            keranjangBelanja = []; 
            document.getElementById("input-bayar").value = "";
            muatDataAwal();
          } else { 
            showErrorModal("Gagal Menyimpan", res.message); 
          }
        }).catch(err => {
          console.error(err);
          showErrorModal("Error", "Gagal mengirim data. Pastikan setelan Deploy akses ke 'Anyone'.");
        });
    }

    function formatRupiah(n) { return "Rp " + Number(n).toLocaleString('id-ID'); }
    function showLoadingModal(t, m) { document.getElementById("modal-icon").innerHTML = '<i class="fa-solid fa-circle-notch animate-spin text-indigo-600"></i>'; document.getElementById("modal-title").innerText = t; document.getElementById("modal-message").innerText = m; document.getElementById("modal-btn-close").classList.add("hidden"); document.getElementById("modal-status").style.display = "flex"; }
    function showSuccessModal(t, m) { document.getElementById("modal-icon").innerHTML = '<i class="fa-solid fa-circle-check text-emerald-500"></i>'; document.getElementById("modal-title").innerText = t; document.getElementById("modal-message").innerText = m; document.getElementById("modal-btn-close").classList.remove("hidden"); document.getElementById("modal-status").style.display = "flex"; }
    function showErrorModal(t, m) { document.getElementById("modal-icon").innerHTML = '<i class="fa-solid fa-circle-xmark text-rose-500"></i>'; document.getElementById("modal-title").innerText = t; document.getElementById("modal-message").innerText = m; document.getElementById("modal-btn-close").classList.remove("hidden"); document.getElementById("modal-status").style.display = "flex"; }
    function tutupModal() { document.getElementById("modal-status").style.display = "none"; }
  </script>
